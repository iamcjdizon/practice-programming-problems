import java.util.*;

public class Easy
{	
	public static void main (String[] args)
	{
		Scanner s = new Scanner (System.in);
		System.out.print("Enter a number: ");
		int a = s.nextInt();
		int b;
		int x, y, z;
		
		for(x = 1; x<=a; x++)
		{
			for (y=1; y<=x; y++)
			{
				System.out.print(y);
			}
			System.out.print(" = ");
			for (z=1; z<=x; z++)
			{
				b=10-z;
				System.out.print(b);
			}
			System.out.print("\n");
		}
		
	}
}
